function sample_uniform_points(rng, n; xmin=-2.0, xmax=2.0, ymin=-2.0, ymax=2.0)
    X = zeros(Float64, 2, n)
    X[1, :] .= rand(rng, n) .* (xmax - xmin) .+ xmin
    X[2, :] .= rand(rng, n) .* (ymax - ymin) .+ ymin
    return to_train_set(X)
end

function boundary_sets(n=40)
    xs = collect(range(-2.0, 2.0; length=n))
    ys = collect(range(-2.0, 2.0; length=n))

    Xbottom = hcat([[x, -2.0] for x in xs]...)
    Xtop = hcat([[x, 2.0] for x in xs]...)
    Xleft = hcat([[-2.0, y] for y in ys]...)
    Xright = hcat([[2.0, y] for y in ys]...)

    return [to_train_set(Xbottom),
            to_train_set(Xtop),
            to_train_set(Xleft),
            to_train_set(Xright)]
end

pde_residual_fun = symprob.loss_functions.datafree_pde_loss_functions[1]
bc_residual_funs = symprob.loss_functions.datafree_bc_loss_functions


function total_loss(theta, Xint, Xbc_sets; lambda_bc=10.0)
    r_pde = vec(pde_residual_fun(Xint, theta))
    l_pde = mean(abs2, r_pde)

    bc_terms = [mean(abs2, vec(bc_fun(Xbc, theta))) for (bc_fun, Xbc) in zip(bc_residual_funs, Xbc_sets)]
    l_bc = mean(bc_terms)

    return l_pde + lambda_bc * l_bc
end

function l2_error(theta; xmin=-2.0, xmax=2.0, ymin=-2.0, ymax=2.0, n_eval=81)
    xs = range(xmin, xmax; length=n_eval)
    ys = range(ymin, ymax; length=n_eval)

    u_pred = [u_scalar([x, y], theta) for x in xs for y in ys]
    u_exact = [analytic_sol_func(x, y) for x in xs for y in ys]

    return sqrt(mean((u_pred .- u_exact).^2))
end

function build_sampling_distribution(Xpool, theta; eps=1e-8, alpha=2.0)
    a = abs.(vec(pde_residual_fun(Xpool, theta))) .+ eps
    w = a .^ alpha
    g = w ./ sum(w)
    return a, g
end

function sample_new_points(rng, theta; pool_size=200, n_add=20, eps=1e-8, alpha=2.0)
    Xpool = sample_uniform_points(rng, pool_size)
    a, g = build_sampling_distribution(Xpool, theta; eps=eps, alpha=alpha)

    idx = sample(rng, 1:size(Xpool, 2), ProbabilityWeights(g), n_add; replace=false)
    Xnew = Xpool[:, idx]

    return Xnew, Xpool, a, g
end

function train_adam_block!(
    theta,
    opt_state,
    Xint,
    Xbc_sets;
    iters,
    record_every,
    lambda_bc,
    iter_hist,
    loss_hist,
    l2_hist,
    total_iters_ref
)
    loss_closure = θ -> total_loss(θ, Xint, Xbc_sets; lambda_bc=lambda_bc)

    for _ in 1:iters
        grads = Zygote.gradient(loss_closure, theta)[1]
        opt_state, theta = Optimisers.update(opt_state, theta, grads)

        total_iters_ref[] += 1

        if (total_iters_ref[] % record_every == 0)
            current_loss = loss_closure(theta)
            current_l2 = l2_error(theta)

            push!(iter_hist, total_iters_ref[])
            push!(loss_hist, current_loss)
            push!(l2_hist, current_l2)
        end
    end

    # чтобы конец блока тоже записался, даже если iters не делится на record_every
    if isempty(iter_hist) || iter_hist[end] != total_iters_ref[]
        current_loss = loss_closure(theta)
        current_l2 = l2_error(theta)

        push!(iter_hist, total_iters_ref[])
        push!(loss_hist, current_loss)
        push!(l2_hist, current_l2)
    end

    return theta, opt_state
end

function adaptive_train!(theta0;
    rng,
    n_init,
    n_bc,
    rounds,
    iters_per_round,
    record_every,
    pool_size,
    n_add,
    lr,
    lambda_bc,
    eps,
    alpha
)
    Xint = sample_uniform_points(rng, n_init)
    Xbc_sets = boundary_sets(n_bc)

    theta = deepcopy(theta0)

    iter_hist = Int[]
    loss_hist = Float64[]
    l2_hist = Float64[]
    added_hist = Matrix{Float64}[]
    pool_hist = Matrix{Float64}[]
    g_hist = Vector{Float64}[]
    total_iters_ref = Ref(0)

    opt_rule = Optimisers.Adam(lr)
    opt_state = Optimisers.setup(opt_rule, theta)

    for k in 1:rounds
        println("=== Adaptive round $k ===")

        theta, opt_state = train_adam_block!(
            theta,
            opt_state,
            Xint,
            Xbc_sets;
            iters=iters_per_round,
            record_every=record_every,
            lambda_bc=lambda_bc,
            iter_hist=iter_hist,
            loss_hist=loss_hist,
            l2_hist=l2_hist,
            total_iters_ref=total_iters_ref
        )

        current_loss = total_loss(theta, Xint, Xbc_sets; lambda_bc=lambda_bc)
        current_l2 = l2_error(theta)

        Xnew, Xpool, a, g = sample_new_points(
            rng,
            theta;
            pool_size=pool_size,
            n_add=n_add,
            eps=eps,
            alpha=alpha
        )

        Xint = hcat(Xint, Xnew)

        push!(added_hist, Array(Xnew))
        push!(pool_hist, Array(Xpool))
        push!(g_hist, collect(g))

        println("loss = $current_loss")
        println("L2 error = $current_l2")
        println("interior points = $(size(Xint, 2))")

        
    end

    return theta, Xint, Xbc_sets, iter_hist, loss_hist, l2_hist, added_hist, pool_hist, g_hist
end
